fetcher関数の確認 useSWRには通常、fetcher関数が必要です。デフォルトのfetcherを使用している場合は問題ありませんが、明示的に定義することをおすすめします：
typescript

Copy
const fetcher = (url: string) => fetch(url).then(res => res.json());

export function useCompetitions() {
  const { data: competitions, error, mutate } = useSWR<Competition[]>(
    "/api/competitions", 
    fetcher,
    {
      revalidateOnFocus: true,
      revalidateOnMount: true,
      dedupingInterval: 0,
      shouldRetryOnError: true,
      errorRetryCount: 3
    }
  );
  
  return {
    competitions,
    isLoading: !error && !competitions,
    error,
    mutate
  };
}

export function useCompetitions() {
  const { data: competitions, error, mutate } = useSWR<Competition[]>(

        console.error('Competitions fetch error:', error);
      },
      onSuccess: (data) => {
        console.log('Competitions data:', data);
      }
    }
  );
  
  return {
    competitions,
    isLoading: !error && !competitions,
    error,
    mutate
  };
}
コンポーネントでの使用例 データ取得と表示の例：
typescript

Copy
function CompetitionList() {
  const { competitions, isLoading, error } = useCompetitions();

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error loading competitions</div>;
  if (!competitions || competitions.length === 0) return <div>No competitions found</div>;

  return (
    <div>
      {competitions.map((competition) => (
        <div key={competition.id}>
          {/* 大会情報を表示 */}
        </div>
      ))}
    </div>
  );
}
